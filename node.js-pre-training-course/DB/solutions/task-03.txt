

SQL QUERIES EXECUTED
      

FILTERING BY STATUS

Select all todos with status 'active'
SELECT * 
FROM todos t 
WHERE t.status = 'active';

Select all todos with status 'completed'
SELECT * 
FROM todos t 
WHERE t.status = 'completed';


SORTING BY CREATION DATE

Select all todos sorted by creation date in ascending order (oldest first)
SELECT *
FROM todos t 
ORDER BY created_at ASC;

Select all todos sorted by creation date in descending order (newest first)
SELECT *
FROM todos t 
ORDER BY created_at DESC;


SEARCH BY KEYWORD

Select all todos where title contains the keyword 'milk'
SELECT *
FROM todos t 
WHERE title LIKE '%milk%';

QUERY EXPLANATIONS
      

WHERE clause: Filters rows based on specified conditions 
Example: WHERE t.status = 'active' returns only todos with active status

ORDER BY clause: Sorts result set by specified column(s) 
ASC (ascending): Sorts from lowest to highest, oldest to newest
DESC (descending): Sorts from highest to lowest, newest to oldest

LIKE operator: Performs pattern matching on text fields 
% wildcard: Matches any sequence of characters (zero or more)
Example: '%milk%' matches 'buy milk'
